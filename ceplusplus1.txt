#include <iostream>
#include <vector>
#include <cmath>
#include <chrono>

// =======================
// ZADANIE 1 – SZESCiAN
// =======================
bool isCube(long long n) {
    for (long long i = -10000; i <= 10000; i++) {
        if (i * i * i == n) return true;
    }
    return false;
}

// =======================
// ZADANIE 2 – ARMSTRONG
// =======================
bool isArmstrong(long long n) {
    if (n < 0) return false;

    long long temp = n;
    int digits = 0;

    while (temp > 0) {
        digits++;
        temp /= 10;
    }

    temp = n;
    long long sum = 0;

    while (temp > 0) {
        int d = temp % 10;

        long long power = 1;
        for (int i = 0; i < digits; i++) {
            power *= d;
        }

        sum += power;
        temp /= 10;
    }

    return sum == n;
}

std::vector<long long> findArmstrongNumbers(long long limit) {
    std::vector<long long> v;

    for (long long i = 0; i <= limit; i++) {
        if (isArmstrong(i)) {
            v.push_back(i);
        }
    }

    return v;
}

// =======================
// ZADANIE 3 – DWUMIAN
// =======================
unsigned long long binomialRec(int n, int k) {
    if (k == 0 || k == n) return 1;
    return binomialRec(n - 1, k - 1) + binomialRec(n - 1, k);
}

unsigned long long binomialIter(int n, int k) {
    if (k > n) return 0;

    unsigned long long result = 1;
    for (int i = 1; i <= k; i++) {
        result = result * (n - i + 1) / i;
    }
    return result;
}

// =======================
// ZADANIE 3 – NWD
// =======================
long long gcdRec(long long a, long long b) {
    if (b == 0) return a;
    return gcdRec(b, a % b);
}

long long gcdIter(long long a, long long b) {
    while (b != 0) {
        long long temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

// =======================
// MAIN
// =======================
int main() {
    using namespace std::chrono;

    // ---- ZADANIE 1 ----
    std::cout << "SZESCIANY:\n";
    long long nums[] = {27, -64, 100, 343, -1331};

    for (int i = 0; i < 5; i++) {
        std::cout << nums[i] << ": "
                  << (isCube(nums[i]) ? "TAK" : "NIE") << "\n";
    }

    // ---- ZADANIE 2 ----
    std::cout << "\nARMSTRONG:\n";
    auto a = findArmstrongNumbers(10000);
    for (int i = 0; i < a.size(); i++) {
        std::cout << a[i] << " ";
    }

    // ---- ZADANIE 3 – DWUMIAN ----
    int n = 30, k = 15;

    auto start = high_resolution_clock::now();
    auto r1 = binomialRec(n, k);
    auto end = high_resolution_clock::now();
    auto t1 = duration<double, std::milli>(end - start);

    start = high_resolution_clock::now();
    auto r2 = binomialIter(n, k);
    end = high_resolution_clock::now();
    auto t2 = duration<double, std::milli>(end - start);

    std::cout << "\n\nDWUMIAN:\n";
    std::cout << "Rekurencja: " << r1 << " czas: " << t1.count() << " ms\n";
    std::cout << "Iteracja:   " << r2 << " czas: " << t2.count() << " ms\n";

    // ---- ZADANIE 3 – NWD ----
    long long x = 123456789, y = 987654321;

    start = high_resolution_clock::now();
    auto g1 = gcdRec(x, y);
    end = high_resolution_clock::now();
    auto tg1 = duration<double, std::milli>(end - start);

    start = high_resolution_clock::now();
    auto g2 = gcdIter(x, y);
    end = high_resolution_clock::now();
    auto tg2 = duration<double, std::milli>(end - start);

    std::cout << "\nNWD:\n";
    std::cout << "Rekurencja: " << g1 << " czas: " << tg1.count() << " ms\n";
    std::cout << "Iteracja:   " << g2 << " czas: " << tg2.count() << " ms\n";

    return 0;
}
