#include <iostream>
#include <vector>
#include <cmath>

// =======================
// SZESCian (naiwnie)
// =======================
bool isCube(long long n) {
    for (long long i = -10000; i <= 10000; i++) {
        if (i * i * i == n) {
            return true;
        }
    }
    return false;
}

// =======================
// ARMSTRONG (naiwnie)
// =======================
bool isArmstrong(long long n) {
    if (n < 0) return false;

    long long temp = n;
    int digits = 0;

    // liczenie cyfr
    while (temp > 0) {
        digits++;
        temp /= 10;
    }

    temp = n;
    long long sum = 0;

    // sumowanie
    while (temp > 0) {
        int d = temp % 10;

        long long power = 1;
        for (int i = 0; i < digits; i++) {
            power *= d; // zamiast pow()
        }

        sum += power;
        temp /= 10;
    }

    return sum == n;
}

std::vector<long long> findArmstrongNumbers(long long limit) {
    std::vector<long long> v;

    for (long long i = 0; i <= limit; i++) {
        if (isArmstrong(i)) {
            v.push_back(i);
        }
    }

    return v;
}

// =======================
// DWUMIAN (naiwnie)
// =======================
unsigned long long binomialRecursive(int n, int k) {
    if (k == 0 || k == n) return 1;
    return binomialRecursive(n - 1, k - 1) + binomialRecursive(n - 1, k);
}

// =======================
// NWD (naiwnie)
// =======================
long long gcd(long long a, long long b) {
    // brutalna metoda
    long long min = (a < b) ? a : b;

    for (long long i = min; i >= 1; i--) {
        if (a % i == 0 && b % i == 0) {
            return i;
        }
    }

    return 1;
}

// =======================
// MAIN
// =======================
int main() {
    std::cout << "SZESCIANY:\n";
    long long nums[] = {27, -64, 100, 343, -1331};

    for (int i = 0; i < 5; i++) {
        std::cout << nums[i] << ": " << (isCube(nums[i]) ? "TAK" : "NIE") << "\n";
    }

    std::cout << "\nARMSTRONG:\n";
    std::vector<long long> a = findArmstrongNumbers(10000);
    for (int i = 0; i < a.size(); i++) {
        std::cout << a[i] << " ";
    }

    std::cout << "\n\nDWUMIAN:\n";
    std::cout << binomialRecursive(10, 5) << "\n";

    std::cout << "\nNWD:\n";
    std::cout << gcd(123456789, 987654321) << "\n";

    return 0;
}
